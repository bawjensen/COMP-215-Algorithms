T(0) & = 1 \\ \\
T(n) & = T(n-1) + 1 \\ \\
& = T(n-1-1) + 1 + 1 \\
& = T(n-2) + 2 \\ \\
& = T(n-2-1) + 2 + 1 \\
& = T(n-3) + 3 \\ \\
& = T(n-3-1) + 3 + 1 \\
& = T(n-4) + 4 \\ \\
& = T(n-i) + i \\
& n - i = base case (aka 0 or 1) $\Rightarrow$ i = n \\ \\
& = T(n-n) + n \\
& = 1 + n \\
& = O(n) \\
\\
\end{tabular}
