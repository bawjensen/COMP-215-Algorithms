\fbox{ $T(n)=O(f(n))$ such that $T(n) \le c \cdot f(n)$ for all $n \ge n_{0}$ }
\fbox{
	\begin{tabular}{lll}
	$1$ & constant & Best \\
	$log(n)$ & logarithmic & Typically good \\
	$n$ & linear & Pretty good \\
	$n \cdot log(n)$ & linearithmic & Meh \\
	$n^{2}$ & quadratic & Ok \\
	$n^{3}$ & cubic & Bad \\
	$2^{n}$ & exponential & Very bad \\
	$c^{n}$ & exponential & Very very bad \\
	$n!$ & factorial & Doesn't get worse \\
	\end{tabular}
}
